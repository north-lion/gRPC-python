# Install below Libraries to start gRPC

1:
``
python -m pip install --upgrade pip
``

2:
``
python -m pip install grpcio
``

3:
``
python -m pip install grpcio-tools
``

# Codegen RPC
``
python python/codegen.py
``

# Run Server
``
python python/server.py
``

# Run Client
``
python python/client.py
``
